## 3. Prevent Deleting a Client That Has Related Data (HIGH PRIORITY - Data Integrity)

In `server/storage.ts`, update the `deleteClient` method to check if the client has any related contracts, invoices, or projects before deleting. If related data exists, throw an error instead of deleting.

```typescript
async deleteClient(id: string, userId: string) {
  // Check for related contracts
  const [contractCount] = await db.select({ count: count() }).from(contracts)
    .where(and(eq(contracts.clientId, id), eq(contracts.userId, userId)));
  
  // Check for related invoices
  const [invoiceCount] = await db.select({ count: count() }).from(invoices)
    .where(and(eq(invoices.clientId, id), eq(invoices.userId, userId)));
  
  // Check for related projects
  const [projectCount] = await db.select({ count: count() }).from(projects)
    .where(and(eq(projects.clientId, id), eq(projects.userId, userId)));

  const total = contractCount.count + invoiceCount.count + projectCount.count;
  if (total > 0) {
    throw new Error(`CLIENT_HAS_RELATIONS:${contractCount.count}:${invoiceCount.count}:${projectCount.count}`);
  }

  await db.delete(clients).where(and(eq(clients.id, id), eq(clients.userId, userId)));
  return true;
}
```

## 4. Fix Search + Status Filter Not Working Together (MEDIUM PRIORITY)

In `server/storage.ts`, the `getClients` method builds queries separately for search and status, so they override each other. Fix it to combine both conditions:

```typescript
async getClients(userId: string, search?: string, status?: string) {
  const conditions = [eq(clients.userId, userId)];
  
  if (status && status !== "all") {
    conditions.push(eq(clients.status, status));
  }
  
  if (search) {
    const searchPattern = `%${search}%`;
    conditions.push(
      or(
        like(clients.name, searchPattern),
        like(clients.email, searchPattern),
        like(clients.company, searchPattern)
      )!
    );
  }
  
  return db.select().from(clients)
    .where(and(...conditions))
    .orderBy(desc(clients.createdAt));
}
```

app.get("/api/clients", isAuthenticated, async (req: Request, res: Response) => {
  try {
    const { search, status, page = "1", limit = "20" } = req.query;
    const pageNum = Math.max(1, parseInt(page as string));
    const limitNum = Math.min(100, Math.max(1, parseInt(limit as string)));

    const { data, total } = await storage.getClients(getUserId(req), search as string, status as string, limitNum, offset);
