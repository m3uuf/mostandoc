
أبغاك تبني لي تطبيق ويب كامل اسمه مستندك (Mustanadak) - منصة إدارة أعمال للمستقلين وأصحاب المشاريع الصغيرة.
التقنيات المطلوبة:

Frontend: Next.js 14 (App Router) + TypeScript + Tailwind CSS
UI: shadcn/ui components
Backend: Next.js API Routes + Server Actions
Database: Supabase (PostgreSQL)
Auth: Supabase Auth (email/password + Google login)
File Storage: Supabase Storage (للصور والملفات)
Email: Resend (للإشعارات)
اللغة: عربي بالكامل (RTL) مع دعم التبديل للإنجليزية

تصميم الواجهة:

الاتجاه الافتراضي RTL (عربي)
ألوان احترافية: أزرق داكن (#1B4F72) كلون أساسي، أزرق فاتح (#2E86C1) ثانوي، أخضر (#27AE60) للنجاح
تصميم نظيف وبسيط وحديث
متجاوب بالكامل (Mobile-first)
الخط: IBM Plex Sans Arabic للعربي، Inter للإنجليزي
Sidebar navigation على اليمين (لأنه عربي RTL)
Dark mode و Light mode


1. نظام التسجيل والدخول (Authentication)
صفحة التسجيل (/register):

الحقول: الاسم الكامل، البريد الإلكتروني، كلمة المرور، تأكيد كلمة المرور، رقم الجوال (اختياري)
زر "سجّل حساب جديد"
زر "سجّل بحساب Google"
رابط "عندك حساب؟ سجّل دخول"
تحقق من قوة كلمة المرور (8 أحرف على الأقل، حرف كبير، رقم)
إرسال بريد تأكيد بعد التسجيل
بعد التأكيد، يتوجه المستخدم لصفحة إعداد البروفايل

صفحة إعداد البروفايل (Onboarding - /setup):
تظهر مرة واحدة بعد أول تسجيل:

اختيار username (يكون فريد - هذا رابط صفحته العامة mustanadak.com/username)
رفع صورة شخصية
وصف مختصر (Bio)
مجال العمل (تصميم، برمجة، تسويق، استشارات، تصوير، كتابة، أخرى)
زر "أكمل لاحقاً" + زر "ابدأ الآن"

صفحة تسجيل الدخول (/login):

البريد الإلكتروني + كلمة المرور
زر "دخول بحساب Google"
رابط "نسيت كلمة المرور؟"
رابط "ما عندك حساب؟ سجّل الآن"

نسيت كلمة المرور (/forgot-password):

إدخال البريد الإلكتروني
إرسال رابط إعادة تعيين
صفحة إعادة تعيين كلمة المرور

Middleware:

حماية جميع الصفحات (ما عدا Login, Register, الصفحات العامة، وصفحات البورتفوليو)
Redirect للداشبورد إذا المستخدم مسجل دخول وحاول يدخل Login/Register

sqlCREATE TABLE clients (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  email TEXT,
  phone TEXT,
  company TEXT,
  status TEXT DEFAULT 'active' CHECK (status IN ('active', 'prospect', 'inactive')),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Row Level Security
ALTER TABLE clients ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own clients" ON clients

4. إدارة العقود (Contracts - /dashboard/contracts)
جدول قاعدة البيانات (contracts):
sqlCREATE TABLE contracts (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  client_id UUID REFERENCES clients(id) ON DELETE SET NULL,
  title TEXT NOT NULL,
  description TEXT,
  content TEXT, -- محتوى العقد (Rich Text)
  value DECIMAL(12,2),

  status TEXT DEFAULT 'draft' CHECK (status IN ('draft', 'active', 'completed', 'expired', 'terminated')),

ALTER TABLE contracts ENABLE ROW LEVEL SECURITY;

5. إدارة الفواتير (Invoices - /dashboard/invoices)
جداول قاعدة البيانات:
sqlCREATE TABLE invoices (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  client_id UUID REFERENCES clients(id) ON DELETE SET NULL,
  invoice_number TEXT NOT NULL, -- رقم تسلسلي تلقائي
  status TEXT DEFAULT 'draft' CHECK (status IN ('draft', 'sent', 'paid', 'overdue', 'cancelled')),
  issue_date DATE DEFAULT CURRENT_DATE,
  due_date DATE,
  subtotal DECIMAL(12,2) DEFAULT 0,
  vat_rate DECIMAL(5,2) DEFAULT 15.00,
  vat_amount DECIMAL(12,2) DEFAULT 0,
  total DECIMAL(12,2) DEFAULT 0,
  notes TEXT,
  payment_method TEXT, -- تحويل بنكي، نقدي، إلكتروني
  paid_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE invoice_items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  invoice_id UUID REFERENCES invoices(id) ON DELETE CASCADE,
  description TEXT NOT NULL,
  quantity DECIMAL(10,2) DEFAULT 1,
  unit_price DECIMAL(12,2) NOT NULL,
  total DECIMAL(12,2) NOT NULL,
  sort_order INT DEFAULT 0
);

ALTER TABLE invoices ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own invoices" ON invoices
  FOR ALL USING (auth.uid() = user_id);

ALTER TABLE invoice_items ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own invoice items" ON invoice_items
  FOR ALL USING (
    invoice_id IN (SELECT id FROM invoices WHERE user_id = auth.uid())
  );
صفحة قائمة الفواتير (/dashboard/invoices):

زر "إنشاء فاتورة"
5 تبويبات فلترة: الكل، مسودة، مرسلة، مدفوعة، متأخرة

جدول: رقم الفاتورة، العميل، المبلغ، الحالة (badge ملون)، تاريخ الإصدار، تاريخ الاستحقاق
أيقونات: عرض، تعديل، تحميل PDF، تغيير الحالة

صفحة إنشاء/تعديل فاتورة (/dashboard/invoices/new أو /[id]/edit):

7. الصفحة العامة - بورتفوليو (Public Page - /[username])
جداول قاعدة البيانات:
sqlCREATE TABLE profiles (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE UNIQUE,
  username TEXT UNIQUE NOT NULL,
  full_name TEXT,
  bio TEXT,
  avatar_url TEXT,
  profession TEXT, -- مجال العمل
  location TEXT,
  email_public TEXT, -- إيميل يظهر في الصفحة العامة
  phone_public TEXT,
  website TEXT,
  social_links JSONB DEFAULT '{}', -- {"twitter": "url", "linkedin": "url", "instagram": "url", "whatsapp": "number"}
  theme TEXT DEFAULT 'default', -- اسم الثيم
  primary_color TEXT DEFAULT '#1B4F72',
  is_public BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE services (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  profile_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  description TEXT,
  price DECIMAL(12,2),
  price_type TEXT DEFAULT 'fixed' CHECK (price_type IN ('fixed', 'hourly', 'negotiable')),
  sort_order INT DEFAULT 0,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE portfolio_items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  profile_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  description TEXT,
  image_url TEXT,
  link TEXT,
  category TEXT,
  sort_order INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE contact_messages (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  profile_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  sender_name TEXT NOT NULL,
  sender_email TEXT NOT NULL,
  message TEXT NOT NULL,
  is_read BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Public access for profile pages
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public profiles are viewable" ON profiles
  FOR SELECT USING (is_public = true);
CREATE POLICY "Users can manage own profile" ON profiles
  FOR ALL USING (auth.uid() = user_id);

ALTER TABLE services ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public services are viewable" ON services
  FOR SELECT USING (
    profile_id IN (SELECT id FROM profiles WHERE is_public = true)
  );
CREATE POLICY "Users can manage own services" ON services
  FOR ALL USING (
    profile_id IN (SELECT id FROM profiles WHERE user_id = auth.uid())
  );

-- نفس السياسة لـ portfolio_items و contact_messages
ALTER TABLE portfolio_items ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public portfolio viewable" ON portfolio_items
  FOR SELECT USING (
    profile_id IN (SELECT id FROM profiles WHERE is_public = true)
  );
CREATE POLICY "Users can manage own portfolio" ON portfolio_items
  FOR ALL USING (
    profile_id IN (SELECT id FROM profiles WHERE user_id = auth.uid())
  );

ALTER TABLE contact_messages ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Anyone can send messages" ON contact_messages
  FOR INSERT WITH CHECK (true);
CREATE POLICY "Users can read own messages" ON contact_messages
  FOR SELECT USING (
    profile_id IN (SELECT id FROM profiles WHERE user_id = auth.uid())
  );

9. الإعدادات (Settings - /dashboard/settings)
تبويبات الإعدادات:

الحساب: تعديل الاسم، الإيميل، كلمة المرور، رقم الجوال
بيانات الفوترة: اسم الشركة/النشاط، العنوان، الرقم الضريبي (اختياري)، رفع اللوقو
الإشعارات: تفعيل/تعطيل كل نوع من الإشعارات + إشعارات البريد
المظهر: الوضع الداكن/الفاتح، اللغة (عربي/إنجليزي)


10. صفحة الهبوط (Landing Page - /)
صفحة تسويقية للمنصة تحتوي على:

Hero Section: عنوان جذاب "أدِر أعمالك كلها من مكان واحد" + وصف + زر "ابدأ مجاناً" + صورة/mockup للداشبورد
المميزات: 5 مميزات رئيسية مع أيقونات (إدارة العملاء، العقود، الفواتير، المشاريع، الصفحة العامة)
كيف يعمل: 3 خطوات (سجّل حساب → أضف عملاءك → أدِر أعمالك)
الأسعار: 3 خطط (مجاني، بروفيشنال 49 ر.س/شهر، أعمال 99 ر.س/شهر)
Footer: روابط (عن المنصة، تواصل معنا، سياسة الخصوصية)
تصميم عصري وجذاب


11. ملاحظات تقنية مهمة:
الأمان:

Row Level Security (RLS) على كل الجداول - المستخدم يشوف بياناته فقط
تحقق من المدخلات (Validation) في Frontend و Backend
CSRF protection
Rate limiting على API routes

الأداء:

Server-side rendering للصفحات العامة (SEO)
Client-side للداشبورد
Optimistic updates للتفاعلات السريعة
Image optimization مع Next.js Image
